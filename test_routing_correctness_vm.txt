# Build original 1.2B
g++ -std=c++17 -O3 -DHAS_LINUX_EPOLL -march=native -mavx2 -mavx -msse4.2 -pthread \
    meteor_phase1_2b_syscall_reduction.cpp -o meteor_original_1_2b \
    -luring -lboost_fiber -lboost_context -lboost_system >/dev/null 2>&1

if [ $? -eq 0 ]; then

    # Test with default auto-detect (should use fewer shards than cores)

        echo "Single command test:"
        ORIG_SINGLE_FAILURES=0
        for i in {1..20}; do
            redis-cli -p 6379 set orig_single_$i "orig_value_$i" >/dev/null
            RESULT=$(redis-cli -p 6379 get orig_single_$i)
            if [ "$RESULT" != "orig_value_$i" ]; then
                ORIG_SINGLE_FAILURES=$((ORIG_SINGLE_FAILURES + 1))
                if [ $ORIG_SINGLE_FAILURES -eq 1 ]; then
                    echo "❌ First failure: set orig_single_$i → get returned '$RESULT'"
                fi
            fi
        done
        echo "Original 1.2B single command failures: $ORIG_SINGLE_FAILURES/20"
        
        # Test pipeline correctness  
        echo ""
        echo "Pipeline test:"
        redis-cli -p 6379 --pipe <<EOF >/dev/null
SET orig_pipe_1 "orig_pipe_value_1"
SET orig_pipe_2 "orig_pipe_value_2"
SET orig_pipe_3 "orig_pipe_value_3"
SET orig_pipe_4 "orig_pipe_value_4"
SET orig_pipe_5 "orig_pipe_value_5"
EOF
        
        ORIG_PIPELINE_FAILURES=0

            RESULT=$(redis-cli -p 6379 get orig_pipe_$i)

                fi

        ORIG_SINGLE_FAILURES="N/A"
        ORIG_PIPELINE_FAILURES="N/A"

    kill $ORIG_PID 2>/dev/null && sleep 3
else
    echo "❌ Original 1.2B build failed"

    ORIG_PIPELINE_FAILURES="N/A"
