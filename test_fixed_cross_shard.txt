# Setup
cd /mnt/externalDisk/meteor
export TMPDIR=/dev/shm
pkill -f meteor 2>/dev/null || true && sleep 3

# Build fixed version
echo "🔧 Building FIXED cross-shard version (shared global channels)..."
g++ -std=c++17 -O3 -DHAS_LINUX_EPOLL -march=native -mavx2 -mavx -msse4.2 -pthread meteor_phase1_2b_dragonfly_cross_shard_fixed.cpp -o meteor_fixed_cross_shard -luring -lboost_fiber -lboost_context -lboost_system

if [ $? -ne 0 ]; then
    echo "❌ Build failed"
    exit 1
fi

echo "✅ Fixed cross-shard version built!"

# Start server with cross-shard setup (6C:3S forces cross-shard operations)
echo ""
echo "🧪 Testing FIXED cross-shard (6C:3S) - should work now!"
echo "====================================================="

./meteor_fixed_cross_shard -h 127.0.0.1 -p 6379 -c 6 -s 3 -m 1536 &
SERVER_PID=$!
sleep 5

# Test connectivity
redis-cli -p 6379 ping

# Clear data
redis-cli -p 6379 flushall

# CRITICAL TEST: Cross-shard SET→GET operations
echo ""
echo "🎯 CRITICAL TEST: Cross-shard correctness with shared channels"
echo "============================================================="

FAILURES=0
for i in {1..15}; do
    redis-cli -p 6379 set fixed_key_$i "fixed_value_$i" >/dev/null
    RESULT=$(redis-cli -p 6379 get fixed_key_$i)
    if [ "$RESULT" != "fixed_value_$i" ]; then
        FAILURES=$((FAILURES + 1))
        if [ $FAILURES -eq 1 ]; then
            echo "❌ First failure: set fixed_key_$i → get '$RESULT'"
        fi
    fi
done

echo "Fixed cross-shard failures: $FAILURES/15"

# Test rapid consistency
echo ""
echo "🚀 Rapid consistency test:"
echo "========================="

RAPID_FAILURES=0
for i in {1..5}; do
    redis-cli -p 6379 set rapid_$i "rapid_val_$i" >/dev/null
    RAPID_RESULT=$(redis-cli -p 6379 get rapid_$i)
    if [ "$RAPID_RESULT" != "rapid_val_$i" ]; then
        RAPID_FAILURES=$((RAPID_FAILURES + 1))
    fi
done

echo "Rapid SET→GET failures: $RAPID_FAILURES/5"

# Manual verification
echo ""
echo "🔍 Manual verification:"
echo "======================="

redis-cli -p 6379 set manual_test_1 "manual_value_1"
MANUAL1=$(redis-cli -p 6379 get manual_test_1)
echo "Manual test 1: set manual_test_1 → get '$MANUAL1'"

redis-cli -p 6379 set manual_test_2 "manual_value_2" 
MANUAL2=$(redis-cli -p 6379 get manual_test_2)
echo "Manual test 2: set manual_test_2 → get '$MANUAL2'"

redis-cli -p 6379 set manual_test_3 "manual_value_3"
MANUAL3=$(redis-cli -p 6379 get manual_test_3)
echo "Manual test 3: set manual_test_3 → get '$MANUAL3'"

# Performance test if working
if [ $FAILURES -eq 0 ] && [ $RAPID_FAILURES -eq 0 ]; then

    echo "🎉 ALL TESTS PASSED!"
    echo ""
    echo "🚀 Running performance test..."

    echo ""
    echo "🏆 CROSS-SHARD FIX SUCCESSFUL!"

else
