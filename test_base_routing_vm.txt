# 🧪 VM TEST: Base Phase 1.2B Routing Issue

# ===== SSH TO VM =====
gcloud compute ssh --zone "asia-southeast1-a" "memtier-benchmarking" --tunnel-through-iap --project "<your-gcp-project-id>"

# ===== BUILD BASE PHASE 1.2B =====
cd /mnt/externalDisk/meteor
export TMPDIR=/dev/shm
pkill -f meteor 2>/dev/null || true
sleep 3

g++ -std=c++17 -O3 -DHAS_LINUX_EPOLL -march=native -mavx2 -mavx -msse4.2 -pthread -o meteor_base_1_2b meteor_phase1_2b_syscall_reduction.cpp -luring

# ===== TEST 1: ROUTING BROKEN (12C:4S) =====
echo ""
echo "🔍 TEST 1: Base 1.2B with 12C:4S - Should show routing issue"
echo "============================================================"

./meteor_base_1_2b -h 127.0.0.1 -p 6379 -c 12 -s 4 -m 2048 &
SERVER_PID_1=$!
sleep 5

redis-cli -p 6379 ping
redis-cli -p 6379 flushall

echo "Testing SET followed by GET (expecting routing failure):"

# Test multiple keys to see routing inconsistency
ROUTING_FAILURES=0
for i in {1..20}; do
    KEY="test_key_$i"
    
    redis-cli -p 6379 set $KEY "value_$i" >/dev/null
    GET_RESULT=$(redis-cli -p 6379 get $KEY)
    
    if [ "$GET_RESULT" = "value_$i" ]; then
        echo "✅ Key $KEY: Consistent routing"
    else
        echo "❌ Key $KEY: SET succeeded but GET failed (routing issue!)"
        echo "   SET stored in one core, GET checked different core"
        ROUTING_FAILURES=$((ROUTING_FAILURES + 1))
    fi
done

echo ""
echo "🎯 Routing failures with 12C:4S: $ROUTING_FAILURES/20"

if [ $ROUTING_FAILURES -gt 0 ]; then
    echo "❌ CONFIRMED: Base 1.2B has routing issue with cores > shards"
else
    echo "✅ Surprising: No routing issues found"
fi

kill $SERVER_PID_1 2>/dev/null || true
sleep 3

# ===== TEST 2: ROUTING FIXED (12C:12S) =====
echo ""
echo "🔍 TEST 2: Base 1.2B with 12C:12S - Should fix routing"
echo "======================================================"

./meteor_base_1_2b -h 127.0.0.1 -p 6379 -c 12 -s 12 -m 2048 &
SERVER_PID_2=$!
sleep 5

redis-cli -p 6379 ping
redis-cli -p 6379 flushall

echo "Testing with cores=shards (expecting perfect routing):"

PERFECT_ROUTING_FAILURES=0
for i in {1..20}; do
    KEY="perfect_key_$i"
    
    redis-cli -p 6379 set $KEY "perfect_value_$i" >/dev/null
    GET_RESULT=$(redis-cli -p 6379 get $KEY)
    
    if [ "$GET_RESULT" = "perfect_value_$i" ]; then
        echo "✅ Key $KEY: Perfect routing"
    else
        echo "❌ Key $KEY: Routing still broken even with cores=shards!"
        PERFECT_ROUTING_FAILURES=$((PERFECT_ROUTING_FAILURES + 1))
    fi
done

echo ""
echo "🎯 Routing failures with 12C:12S: $PERFECT_ROUTING_FAILURES/20"

if [ $PERFECT_ROUTING_FAILURES -eq 0 ]; then
    echo "🎉 SOLUTION FOUND: cores=shards fixes routing!"
    echo "   Each core owns exactly one shard → perfect consistency"
else
    echo "❌ cores=shards doesn't fix it → deeper architecture issue"
fi

kill $SERVER_PID_2 2>/dev/null || true
sleep 3

# ===== TEST 3: BENCHMARK FIXED ROUTING =====
echo ""
echo "🔍 TEST 3: Benchmark with fixed routing (12C:12S)"
echo "================================================="

if [ $PERFECT_ROUTING_FAILURES -eq 0 ]; then
    echo "✅ Running benchmark with perfect routing..."
    
    ./meteor_base_1_2b -h 127.0.0.1 -p 6379 -c 12 -s 12 -m 2048 &
    SERVER_PID_3=$!
    sleep 5
    
    # Quick benchmark to verify performance
    memtier_benchmark --server=127.0.0.1 --port=6379 --protocol=redis --clients=20 --threads=8 --pipeline=10 --data-size=64 --key-pattern=R:R --ratio=1:3 --test-time=15
    
    kill $SERVER_PID_3 2>/dev/null || true
else
    echo "⏭️  Skipping benchmark - routing still broken"
fi

# ===== SUMMARY =====
echo ""
echo "🏆 BASE 1.2B ROUTING ANALYSIS COMPLETE"
echo "======================================"
echo ""

echo "12C:4S routing failures: $ROUTING_FAILURES/20"
echo "12C:12S routing failures: $PERFECT_ROUTING_FAILURES/20"
echo ""
echo "CONCLUSION:"
if [ $ROUTING_FAILURES -gt 0 ] && [ $PERFECT_ROUTING_FAILURES -eq 0 ]; then
    echo "✅ cores=shards FIXES the routing issue!"
    echo "   → Base 1.2B confirmed has routing problem"
    echo "   → Solution: Always use cores=shards configuration"
    echo "   → Apply this fix to commercial LRU+TTL version"
elif [ $ROUTING_FAILURES -gt 0 ] && [ $PERFECT_ROUTING_FAILURES -gt 0 ]; then
    echo "❌ cores=shards doesn't fix it → architecture problem"
    echo "   → Need shared cache or proper cross-core access"
else
    echo "✅ No routing issues found → investigate TTL-specific problem"
fi
