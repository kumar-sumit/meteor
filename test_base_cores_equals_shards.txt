    RESULT=$(redis-cli -p 6379 get broken_key_$i)
    if [ "$RESULT" != "broken_value_$i" ]; then
        BROKEN_FAILURES=$((BROKEN_FAILURES + 1))

        fi
    fi
done

    fi
